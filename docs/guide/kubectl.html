kubectl Reference

Comprehensive reference for all kubectl commands supported by Rusternetes. Grouped by category with key flags and examples.

Global Flags

These flags work with any kubectl command:

FlagDescription
--kubeconfigPath to kubeconfig file. Default ~/.kube/config. For Rusternetes: ~/.kube/rusternetes-config.
--contextKubeconfig context to use.
--server / -sAPI server URL (overrides kubeconfig).
--tokenBearer token for authentication.
--insecure-skip-tls-verifySkip TLS certificate verification.
--namespace / -nNamespace scope for the request.
--all-namespaces / -AList across all namespaces.
-o / --outputOutput format: json, yaml, wide, name, custom-columns=..., jsonpath=....
--dry-runnone (default), client, or server. Preview without persisting.
-v / --vVerbosity level (0-9). -v6 shows HTTP requests. -v8 shows request/response bodies.

Resource Management

get

List one or more resources.

        

          
          
          
          terminal
        

        # List pods in current namespace
kubectl get pods

# Wide output with node and IP info
kubectl get pods -o wide

# Watch for changes in real time
kubectl get pods --watch

# Filter by label
kubectl get pods --selector=app=web
kubectl get pods -l app=web,tier=frontend

# JSON output
kubectl get pod my-pod -o json

# YAML output
kubectl get pod my-pod -o yaml

# Custom columns
kubectl get pods -o custom-columns=NAME:.metadata.name,STATUS:.status.phase,NODE:.spec.nodeName

# JSONPath
kubectl get pod my-pod -o jsonpath='{.status.podIP}'

# All namespaces
kubectl get pods -A

# Multiple resource types
kubectl get pods,services,deployments

# Show labels
kubectl get pods --show-labels

# Sort by a field
kubectl get pods --sort-by=.metadata.creationTimestamp

create

Create resources from files or via subcommands.

        

          
          
          
          terminal
        

        # Create from file
kubectl create -f resource.yaml

# Create with dry-run (preview)
kubectl create -f resource.yaml --dry-run=server

create subcommands

        

          
          
          
          terminal
        

        # Workloads
kubectl create deployment web --image=nginx --replicas=3
kubectl create job batch --image=busybox -- echo "done"
kubectl create cronjob cleanup --image=busybox --schedule="0 * * * *" -- cleanup.sh

# Services
kubectl create service clusterip web --tcp=80:8080
kubectl create service nodeport web --tcp=80:8080 --node-port=30080
kubectl create service loadbalancer web --tcp=80:8080
kubectl create service externalname ext --external-name=db.example.com

# Config
kubectl create configmap app-config --from-literal=key=value --from-file=config.yaml
kubectl create secret generic db-creds --from-literal=password=secret
kubectl create secret tls app-tls --cert=cert.pem --key=key.pem
kubectl create secret docker-registry regcred --docker-server=reg.io --docker-username=user --docker-password=pass

# Namespaces
kubectl create namespace staging

# RBAC
kubectl create role pod-reader --verb=get,list,watch --resource=pods
kubectl create clusterrole node-reader --verb=get,list --resource=nodes
kubectl create rolebinding read-pods --role=pod-reader --user=jane
kubectl create clusterrolebinding read-nodes --clusterrole=node-reader --group=devs

# Other
kubectl create serviceaccount my-app
kubectl create token my-app --duration=1h
kubectl create ingress web --rule="app.example.com/=web:80" --class=nginx
kubectl create quota compute --hard=cpu=4,memory=8Gi,pods=20
kubectl create pdb web-pdb --selector=app=web --max-unavailable=1
kubectl create priorityclass high --value=1000000

apply

Apply a configuration to a resource. Creates the resource if it does not exist.

        

          
          
          
          terminal
        

        # Apply from file
kubectl apply -f deployment.yaml

# Apply from directory
kubectl apply -f ./manifests/

# Server-side apply (uses field ownership)
kubectl apply --server-side -f deployment.yaml

# Server-side apply with custom field manager
kubectl apply --server-side --field-manager=my-controller -f deployment.yaml

# Dry-run (server-side validation without persisting)
kubectl apply -f deployment.yaml --dry-run=server

# Force conflicts in server-side apply
kubectl apply --server-side --force-conflicts -f deployment.yaml

delete

Delete resources.

        

          
          
          
          terminal
        

        # Delete by name
kubectl delete pod my-pod

# Delete from file
kubectl delete -f deployment.yaml

# Delete by label
kubectl delete pods -l app=old

# Force delete (skip grace period)
kubectl delete pod my-pod --force --grace-period=0

# Cascade options
kubectl delete deployment web --cascade=foreground   # Wait for children to delete
kubectl delete deployment web --cascade=background  # Delete parent first
kubectl delete deployment web --cascade=orphan      # Leave children running

# Delete all pods in a namespace
kubectl delete pods --all -n staging

replace

Replace a resource from a complete file (must include resourceVersion).

        

          
          
          
          terminal
        

        kubectl replace -f deployment.yaml
kubectl replace --force -f deployment.yaml  # Delete and recreate

patch

Update specific fields without replacing the whole resource.

        

          
          
          
          terminal
        

        # Strategic merge patch (default)
kubectl patch deployment web -p '{"spec":{"replicas":5}}'

# JSON merge patch
kubectl patch deployment web --type=merge -p '{"spec":{"replicas":5}}'

# JSON patch (RFC 6902)
kubectl patch deployment web --type=json \
  -p '[{"op":"replace","path":"/spec/replicas","value":5}]'

# Patch a subresource
kubectl patch deployment web --subresource=status --type=merge \
  -p '{"status":{"availableReplicas":3}}'

edit

Open a resource in your default editor.

        

          
          
          
          terminal
        

        kubectl edit deployment web
kubectl edit svc web -o json  # Edit as JSON
KUBE_EDITOR=nano kubectl edit deployment web

label

Add, update, or remove labels.

        

          
          
          
          terminal
        

        # Add/update labels
kubectl label pods my-pod env=production
kubectl label pods my-pod env=staging --overwrite

# Remove a label
kubectl label pods my-pod env-

# Label all pods matching a selector
kubectl label pods -l app=web version=v2

annotate

Add, update, or remove annotations.

        

          
          
          
          terminal
        

        kubectl annotate pods my-pod description="Main web server"
kubectl annotate pods my-pod description-  # Remove

Pod Operations

run

Create and run a pod (optionally interactive).

        

          
          
          
          terminal
        

        # Run a simple pod
kubectl run nginx --image=nginx:1.27

# Run with environment variables
kubectl run myapp --image=myapp:v1 --env=DB_HOST=db-svc --env=DB_PORT=5432

# Run with labels
kubectl run myapp --image=myapp:v1 --labels=app=web,tier=frontend

# Interactive pod with shell
kubectl run debug --image=busybox -it --rm -- sh

# Run with specific command
kubectl run pi --image=perl --restart=Never -- perl -Mbignum=bpi -wle "print bpi(2000)"

logs

View container logs.

        

          
          
          
          terminal
        

        kubectl logs my-pod
kubectl logs my-pod --follow             # Stream logs
kubectl logs my-pod --tail=50             # Last 50 lines
kubectl logs my-pod --since=30m           # Last 30 minutes
kubectl logs my-pod --timestamps          # Include timestamps
kubectl logs my-pod --container=sidecar  # Specific container
kubectl logs my-pod --previous            # Previous container instance
kubectl logs my-pod --all-containers     # All containers
kubectl logs -l app=web                 # All pods matching label
kubectl logs job/batch                   # Logs from a job's pod

exec

Execute a command in a container.

        

          
          
          
          terminal
        

        kubectl exec my-pod -- ls /app
kubectl exec -it my-pod -- /bin/sh
kubectl exec -it my-pod --container=app -- /bin/bash

attach

Attach to a running container's process.

        

          
          
          
          terminal
        

        kubectl attach my-pod
kubectl attach -it my-pod

cp

Copy files between host and containers.

        

          
          
          
          terminal
        

        # Copy to pod
kubectl cp local-file.txt my-pod:/tmp/remote-file.txt

# Copy from pod
kubectl cp my-pod:/var/log/app.log ./app.log

# Copy from a specific container
kubectl cp my-pod:/data ./backup/ --container=app

port-forward

Forward local ports to a pod or service.

        

          
          
          
          terminal
        

        kubectl port-forward my-pod 8080:80
kubectl port-forward svc/web 8080:80
kubectl port-forward my-pod 8080:80 9090:9090
kubectl port-forward --address=0.0.0.0 my-pod 8080:80

debug

Create debug containers for troubleshooting.

        

          
          
          
          terminal
        

        kubectl debug my-pod -it --image=busybox
kubectl debug my-pod -it --image=busybox --target=app
kubectl debug my-pod -it --copy-to=debug-pod --image=ubuntu
kubectl debug node/node-1 -it --image=ubuntu

Cluster Operations

top

Show resource usage (CPU/memory) for nodes and pods.

        

          
          
          
          terminal
        

        kubectl top nodes
kubectl top pods
kubectl top pods --containers        # Show per-container usage
kubectl top pods -A --sort-by=cpu  # All namespaces, sorted by CPU

drain

Prepare a node for maintenance by evicting pods.

        

          
          
          
          terminal
        

        kubectl drain node-1 --ignore-daemonsets --delete-emptydir-data
kubectl drain node-1 --force  # Include unmanaged pods

cordon / uncordon

Mark a node as unschedulable or re-enable scheduling.

        

          
          
          
          terminal
        

        kubectl cordon node-1    # Mark unschedulable
kubectl uncordon node-1  # Mark schedulable

taint

Add or remove taints on nodes.

        

          
          
          
          terminal
        

        # Add a taint
kubectl taint nodes node-1 dedicated=gpu:NoSchedule

# Remove a taint
kubectl taint nodes node-1 dedicated=gpu:NoSchedule-

# Overwrite
kubectl taint nodes node-1 key=value:NoExecute --overwrite

events

View cluster events.

        

          
          
          
          terminal
        

        kubectl events
kubectl events --watch
kubectl events --types=Warning
kubectl events -A  # All namespaces

Deployment Management

rollout

Manage rollouts for Deployments, DaemonSets, and StatefulSets.

        

          
          
          
          terminal
        

        # Check rollout status
kubectl rollout status deployment/web

# View rollout history
kubectl rollout history deployment/web
kubectl rollout history deployment/web --revision=3

# Undo (rollback)
kubectl rollout undo deployment/web
kubectl rollout undo deployment/web --to-revision=2

# Restart (triggers new rollout)
kubectl rollout restart deployment/web

# Pause / resume
kubectl rollout pause deployment/web
kubectl rollout resume deployment/web

scale

Change the replica count.

        

          
          
          
          terminal
        

        kubectl scale deployment/web --replicas=5
kubectl scale statefulset/db --replicas=3
kubectl scale rs/web-abc123 --replicas=0

expose

Create a Service for a resource.

        

          
          
          
          terminal
        

        kubectl expose deployment web --port=80 --target-port=8080
kubectl expose deployment web --port=80 --type=NodePort
kubectl expose deployment web --port=80 --type=LoadBalancer

autoscale

Create an HPA for a resource.

        

          
          
          
          terminal
        

        kubectl autoscale deployment web --min=2 --max=20 --cpu-percent=70

Auth

auth can-i

Check whether an action is allowed.

        

          
          
          
          terminal
        

        # Can I do this?
kubectl auth can-i get pods
kubectl auth can-i create deployments --namespace=production
kubectl auth can-i "*" "*"                 # Full admin check
kubectl auth can-i list pods --all-namespaces

# Check as another user
kubectl auth can-i get secrets --as=jane
kubectl auth can-i get secrets --as=system:serviceaccount:default:my-app

auth whoami

Show the current authenticated identity.

        

          
          
          
          terminal
        

        kubectl auth whoami

auth reconcile

Reconcile RBAC resources from a file.

        

          
          
          
          terminal
        

        kubectl auth reconcile -f rbac.yaml

Info & Discovery

explain

Show documentation for a resource or field.

        

          
          
          
          terminal
        

        kubectl explain pods
kubectl explain pods.spec.containers
kubectl explain deployment.spec.strategy
kubectl explain pods --recursive  # Show all fields

api-resources

List all available resource types.

        

          
          
          
          terminal
        

        kubectl api-resources
kubectl api-resources --namespaced                  # Namespaced only
kubectl api-resources --namespaced=false            # Cluster-scoped only
kubectl api-resources --verbs=list,get              # With specific verbs
kubectl api-resources --api-group=apps             # From specific API group

api-versions

List all available API versions.

        

          
          
          
          terminal
        

        kubectl api-versions

cluster-info

Show cluster endpoint information.

        

          
          
          
          terminal
        

        kubectl cluster-info
kubectl cluster-info dump  # Full cluster state for debugging

version

Show client and server versions.

        

          
          
          
          terminal
        

        kubectl version
kubectl version --client  # Client only (no server connection needed)

describe

Show detailed resource information including events.

        

          
          
          
          terminal
        

        kubectl describe pod my-pod
kubectl describe node node-1
kubectl describe svc web
kubectl describe deployment web

Config

Manage kubeconfig files, contexts, and clusters.

        

          
          
          
          terminal
        

        # View config
kubectl config view
kubectl config view --minify   # Only current context
kubectl config view --flatten  # Inline certificate data

# Contexts
kubectl config current-context
kubectl config get-contexts
kubectl config use-context rusternetes

# Set values
kubectl config set current-context rusternetes
kubectl config set-cluster rusternetes --server=https://localhost:6443
kubectl config set-credentials admin --token=mytoken

# Unset values
kubectl config unset users.admin

Other Commands

diff

Show differences between the live state and a local file.

        

          
          
          
          terminal
        

        kubectl diff -f deployment.yaml

wait

Wait for a specific condition on a resource.

        

          
          
          
          terminal
        

        # Wait for pod to be ready
kubectl wait --for=condition=Ready pod/my-pod --timeout=120s

# Wait for deployment to be available
kubectl wait --for=condition=Available deployment/web

# Wait for deletion
kubectl wait --for=delete pod/my-pod

# Wait for a specific JSON path value
kubectl wait --for=jsonpath='{.status.phase}'=Running pod/my-pod

certificate

Manage certificate signing requests.

        

          
          
          
          terminal
        

        kubectl certificate approve my-csr
kubectl certificate deny my-csr

completion

Generate shell completion scripts.

        

          
          
          
          terminal
        

        # Bash
source <(kubectl completion bash)

# Zsh
source <(kubectl completion zsh)

# Fish
kubectl completion fish | source

kustomize

Process kustomization files.

        

          
          
          
          terminal
        

        kubectl kustomize ./overlays/production/
kubectl apply -k ./overlays/production/

proxy

Run a local proxy to the API server.

        

          
          
          
          terminal
        

        kubectl proxy
kubectl proxy --port=8001
# Then access: http://localhost:8001/api/v1/namespaces/default/pods

plugin list

List installed kubectl plugins.

        

          
          
          
          terminal
        

        kubectl plugin list

set

Update specific fields on resources.

        

          
          
          
          terminal
        

        # Update container image
kubectl set image deployment/web nginx=nginx:1.28

# Set environment variables
kubectl set env deployment/web DB_HOST=db-svc

# Set resource requests/limits
kubectl set resources deployment/web -c=nginx --requests=cpu=100m,memory=128Mi

# Set a service account
kubectl set serviceaccount deployment/web my-sa

Output Formats

FormatFlagDescription
Table(default)Human-readable table with key columns.
Wide-o wideTable with additional columns (IP, node, etc.).
JSON-o jsonFull JSON representation.
YAML-o yamlFull YAML representation.
Name-o nameJust resource/name (e.g., pod/my-pod).
Custom Columns-o custom-columns=...Custom table with JSONPath columns.
JSONPath-o jsonpath='{...}'Extract specific fields with JSONPath expressions.
Go Template-o go-template='...'Format output with Go templates.

JSONPath Examples

        

          
          
          
          terminal
        

        # Get pod IP
kubectl get pod my-pod -o jsonpath='{.status.podIP}'

# Get all pod names
kubectl get pods -o jsonpath='{.items[*].metadata.name}'

# Get container images
kubectl get pods -o jsonpath='{range .items[*]}{.metadata.name}{"\t"}{.spec.containers[*].image}{"\n"}{end}'

# Get node IPs
kubectl get nodes -o jsonpath='{.items[*].status.addresses[?(@.type=="InternalIP")].address}'
